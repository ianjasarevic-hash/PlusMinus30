// Gallery slider functionality function initGallerySlider() { const slider = document.querySelector('.gallery-slider'); const track = document.getElementById('galleryTrack'); const slides = document.querySelectorAll('.gallery-slide'); const prevBtn = document.getElementById('galleryPrev'); const nextBtn = document.getElementById('galleryNext'); if (!track || slides.length === 0) return; let currentIndex = 0; const totalSlides = slides.length; let isDragging = false; let dragStarted = false; let dragStartedAt = 0; let wasDragging = false; let dragStartX = 0; const dragDelay = 200; let dragTimeout = null; function goToSlide(index) { if (index < 0) index = totalSlides - 1; if (index >= totalSlides) index = 0; slides[currentIndex].classList.remove('active'); currentIndex = index; slides[currentIndex].classList.add('active'); } prevBtn.addEventListener('click', function(e) { e.preventDefault(); e.stopPropagation(); goToSlide(currentIndex - 1); }); nextBtn.addEventListener('click', function(e) { e.preventDefault(); e.stopPropagation(); goToSlide(currentIndex + 1); }); // Drag functionality slider.addEventListener('mousedown', function(e) { e.preventDefault(); slider.style.cursor = 'grabbing'; wasDragging = false; dragStarted = false; dragStartX = e.clientX; dragStartedAt = Date.now(); dragTimeout = setTimeout(function() { dragStarted = true; isDragging = true; slider.classList.add('dragging'); }, dragDelay); }); slider.addEventListener('mousemove', function(e) { if (!dragStarted) { return; } if (!isDragging) return; const diff = e.clientX - dragStartX; const absDiff = Math.abs(diff); const maxDrag = 200; const percent = Math.min(absDiff / maxDrag, 1); // Reset non-active slides slides.forEach(function(slide) { if (slide.classList.contains('active')) return; slide.style.transition = 'none'; slide.style.opacity = '0'; slide.style.visibility = 'hidden'; }); // Show neighbor slide let neighborIndex; if (diff < 0) { neighborIndex = (currentIndex + 1) % totalSlides; } else if (diff > 0) { neighborIndex = (currentIndex - 1 + totalSlides) % totalSlides; } if (neighborIndex !== undefined) { slides[neighborIndex].style.opacity = percent * 0.8; slides[neighborIndex].style.visibility = 'visible'; slides[neighborIndex].style.zIndex = '5'; } }); slider.addEventListener('mouseup', function(e) { const dragDuration = Date.now() - dragStartedAt; if (dragTimeout) { clearTimeout(dragTimeout); dragTimeout = null; } slider.classList.remove('dragging'); slider.style.cursor = 'grab'; // Reset slide styles slides.forEach(function(slide) { slide.style.transition = ''; slide.style.opacity = ''; slide.style.visibility = ''; slide.style.zIndex = ''; }); if (!dragStarted) { return; } dragStarted = false; isDragging = false; const diff = e.clientX - dragStartX; const absDiff = Math.abs(diff); if (absDiff > 20) { wasDragging = true; if (diff < 0) { goToSlide((currentIndex + 1) % totalSlides); } else if (diff > 0) { goToSlide((currentIndex - 1 + totalSlides) % totalSlides); } } }); slider.addEventListener('mouseleave', function() { if (dragTimeout) { clearTimeout(dragTimeout); dragTimeout = null; } slider.classList.remove('dragging'); slider.style.cursor = 'grab'; dragStarted = false; isDragging = false; slides.forEach(function(slide) { slide.style.transition = ''; slide.style.opacity = ''; slide.style.visibility = ''; slide.style.zIndex = ''; }); }); slider.style.cursor = 'grab'; // Touch support let touchStartX = 0; let touchStartY = 0; let touchEndX = 0; let touchEndY = 0; slider.addEventListener('touchstart', function(e) { touchStartX = e.changedTouches[0].screenX; touchStartY = e.changedTouches[0].screenY; }, { passive: true }); slider.addEventListener('touchend', function(e) { touchEndX = e.changedTouches[0].screenX; touchEndY = e.changedTouches[0].screenY; const diffX = touchStartX - touchEndX; const diffY = Math.abs(touchStartY - touchEndY); const swipeThreshold = 50; // Only change slide if horizontal movement is greater than vertical if (Math.abs(diffX) > swipeThreshold && Math.abs(diffX) > diffY) { if (diffX > 0) { goToSlide(currentIndex + 1); } else { goToSlide(currentIndex - 1); } } }, { passive: true }); // Keyboard navigation document.addEventListener('keydown', function(e) { if (e.key === 'ArrowLeft') { goToSlide(currentIndex - 1); } else if (e.key === 'ArrowRight') { goToSlide(currentIndex + 1); } }); }
